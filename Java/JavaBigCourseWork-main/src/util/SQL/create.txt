CREATE TABLE building (
    id INT AUTO_INCREMENT PRIMARY KEY,
    楼宇名称 VARCHAR(255) NOT NULL,
    楼宇编号 VARCHAR(50) NOT NULL
);

CREATE TABLE Resident (
    id INT AUTO_INCREMENT PRIMARY KEY,
    名称 VARCHAR(255) NOT NULL,
    所属楼宇 INT,
    门牌号 VARCHAR(50) NOT NULL,
    进入密码 VARCHAR(255) NOT NULL,
    FOREIGN KEY (所属楼宇) REFERENCES building(id)
);

CREATE TABLE People (
    id INT AUTO_INCREMENT PRIMARY KEY,
    姓名 VARCHAR(255) NOT NULL,
    是否业主 BOOLEAN NOT NULL,
    身份号 VARCHAR(50) NOT NULL,
    手机号 VARCHAR(20) NOT NULL,
    进入时间 DATETIME NOT NULL,
    离开时间 DATETIME
);
